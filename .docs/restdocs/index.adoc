== 빠른 시작

=== 1. DocsTest 작성

[source,kotlin]
----
class MyControllerDocsTest : RestDocsSupport() {

    private val myFacade: MyFacade = mock()

    override fun initController(): Any = MyController(myFacade)

    @Test
    fun `get my resource`(): Unit {
        // given
        given(myFacade.findById(1L)).willReturn(MyResource(1L, "name"))

        // when & then
        mockMvc.perform(get("/api/resources/{id}", 1L))
            .andExpect(status().isOk)
            .andDo(
                restDocs.document(
                    pathParameters(
                        parameterWithName("id").description("리소스 ID"),
                    ),
                    responseFields(*responseCommonFieldsSubsection()),
                    dataResponseFields(
                        "id" type NUMBER means "리소스 ID",
                        "name" type STRING means "리소스 이름",
                    ),
                )
            )
    }
}
----
